docker network create jenkins - stworzenie sieci dla kontenera

docker run \                                                           
          --name jenkins-docker \   
          --rm \
          --detach \
          --privileged \     
          --network jenkins \                  
          --network-alias docker \              
          --env DOCKER_TLS_CERTDIR=/certs \
          --volume jenkins-docker-certs:/certs/client \
          --volume jenkins-data:/var/jenkins_home \
          --publish 2376:2376 \                           
          docker:dind \                
          --storage-driver overlay2

^ uruchomienie obrazu docker:dind

docker build -t myjenkins-blueocean:2.332.1-1 . - zbudowanie nowego obrazu

docker run \                                                           
          --name jenkins-blueocean \
          --rm \
          --detach \
          --network jenkins \                  
          --env DOCKER_HOST=tcp://docker:2376 \ 
          --env DOCKER_CERT_PATH=/certs/client \
          --env DOCKER_TLS_VERIFY=1 \      
          --publish 8080:8080 \                        
          --publish 50000:50000 \                  
          --volume jenkins-data:/var/jenkins_home \       
          --volume jenkins-docker-certs:/certs/client:ro \
          myjenkins-blueocean:2.332.1-1 

^ uruchomienie kontenera

docker commit "istniejacy kontener" "nazwa uzytkownia/nazwa respo:nazwa taga" - zatwierdzenie zmian

docker push "nazwa uzytkownia/nazwa respo:nazwa taga" - opublikowanie obrazu w serwise Docker Hub
