2 skopiowanie git hooka sprawdzającego czy commit zaczyna się od ŁD400222

#!/bin/sh

COMMIT=`cat $1`
REGEX="^ŁD400222"
echo $COMMIT
if ! [[ $COMMIT =~ $REGEX ]]; then
    echo '# Commit message should start with ŁD400222'
    exit 1;
fi
